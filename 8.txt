$MOD51

ORG 0000H
LJMP START

ORG 0003H
LJMP ISR_INT0

ORG 0030H

START:
SETB IT0
SETB EX0
SETB EA
MOV P1,#0FFH
MOV P2,#00H
CLR F0

MAIN:
SETB P3.0
NOP
CLR P3.0

WAIT_EOC:
JNB P3.1,WAIT_EOC

MOV A,P1
JB F0,OFFSET

MOV P2,A
SJMP MAIN

OFFSET:
SUBB A,#128
MOV P2,A
SJMP MAIN

ISR_INT0:
CPL F0
RETI

END

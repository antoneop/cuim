#include <8051.h>

void main() {
    unsigned char i, j;
    unsigned char mode = 0;        
    unsigned char last_state = 1;
    unsigned char break_flag = 0;
    
   
    unsigned char massiv[11] = {
        0xC0, // 0
        0xF9, // 1
        0xA4, // 2
        0xB0, // 3
        0x99, // 4
        0x92, // 5
        0x82, // 6
        0xF8, // 7
        0x80, // 8
        0x90, // 9
        0xFF  // off
    };
    

    unsigned char nechet[5] = {1, 3, 5, 7, 9};
    
    P1 = massiv[10];
    
    while(1) {
        break_flag = 0;
        
      
        if(mode == 0) {
            for(i = 0; i < 5; i++) {
                P2 = massiv[nechet[i]];
                
                for(j = 0; j < 100; j++) {
                    
                    if(((P3 & 0x01) == 0) && last_state == 1) {
                       
                        for(j = 0; j < 50; j++);
                        if(((P3 & 0x01) == 0)) {
                            mode = !mode;
                            break_flag = 1;
                            
                            while(((P3 & 0x01) == 0));
                            last_state = 1;
                            break;
                        }
                    }
                    last_state = (P3 & 0x01);
                }
                if(break_flag) break;
            }
        }
        
        else {
            for(i = 0; i < 10; i++) {
                P2 = massiv[i];
              
                for(j = 0; j < 100; j++) {
                   
                    if(((P3 & 0x01) == 0) && last_state == 1) {
                        
                        for(j = 0; j < 50; j++);
                        if(((P3 & 0x01) == 0)) {
                            mode = !mode;
                            break_flag = 1;
                            
                            while(((P3 & 0x01) == 0));
                            last_state = 1;
                            break;
                        }
                    }
                    last_state = (P3 & 0x01);
                }
                if(break_flag) break;
            }
        }
    }
}
